export function NMTCommandReference() {
  const [deviceFilter, setDeviceFilter] = useState(null);
  const [sortKey, setSortKey] = useState('value');
  const [sortDir, setSortDir] = useState('asc');

  const deviceKeys = ['pm', 'pg4', 'bp2'];
  const deviceLabels = { pm: 'PM', pg4: 'PG4', bp2: 'BP2' };

  function toggleSort(key) {
    if (sortKey === key) {
      setSortDir(d => d === 'asc' ? 'desc' : 'asc');
    } else {
      setSortKey(key);
      setSortDir('asc');
    }
  }

  const filtered = useMemo(() => {
    let rows = deviceFilter
      ? nmtCommands.filter(r => r[deviceFilter])
      : nmtCommands;
    return [...rows].sort((a, b) => {
      let av = a[sortKey] ?? '';
      let bv = b[sortKey] ?? '';
      if (sortKey === 'value') {
        av = parseInt(av, 16);
        bv = parseInt(bv, 16);
      } else {
        av = av.toLowerCase();
        bv = bv.toLowerCase();
      }
      if (av < bv) return sortDir === 'asc' ? -1 : 1;
      if (av > bv) return sortDir === 'asc' ? 1 : -1;
      return 0;
    });
  }, [deviceFilter, sortKey, sortDir]);

  const SortIndicator = ({ col }) =>
    sortKey !== col
      ? <span style={{ opacity: 0.3, marginLeft: 4 }}>↕</span>
      : <span style={{ marginLeft: 4 }}>{sortDir === 'asc' ? '↑' : '↓'}</span>;

  const Check = ({ active }) => (
    <span style={{ color: active ? 'var(--ifm-color-success, #2da44e)' : 'var(--ifm-color-secondary-darkest, #aaa)' }}>
      {active ? '✓' : '—'}
    </span>
  );

  const btnStyle = (active) => ({
    padding: '4px 12px',
    borderRadius: '4px',
    border: '1px solid var(--ifm-color-primary)',
    background: active ? 'var(--ifm-color-primary)' : 'transparent',
    color: active ? '#fff' : 'var(--ifm-color-primary)',
    cursor: 'pointer',
    fontSize: '0.85rem',
    fontWeight: '600',
  });

  const thStyle = { cursor: 'pointer', userSelect: 'none', whiteSpace: 'nowrap' };

  return (
    <div>
      {/* Filter controls */}
      <div style={{ display: 'flex', flexWrap: 'wrap', gap: '12px', alignItems: 'center', marginBottom: '20px' }}>
        <div style={{ display: 'flex', gap: '4px', alignItems: 'center' }}>
          <span style={{ fontSize: '0.85rem', opacity: 0.7 }}>Filter by device:</span>
          {deviceKeys.map(key => (
            <button key={key} onClick={() => setDeviceFilter(deviceFilter === key ? null : key)} style={btnStyle(deviceFilter === key)}>
              {deviceLabels[key]}
            </button>
          ))}
          {deviceFilter && (
            <button
              onClick={() => setDeviceFilter(null)}
              style={{ padding: '4px 8px', borderRadius: '4px', border: '1px solid #ccc', background: 'transparent', cursor: 'pointer', fontSize: '0.8rem', opacity: 0.6 }}
            >✕ Clear</button>
          )}
        </div>
        <span style={{ fontSize: '0.8rem', opacity: 0.5, marginLeft: 'auto' }}>
          {filtered.length} command{filtered.length !== 1 ? 's' : ''}
          {deviceFilter ? ` on ${deviceLabels[deviceFilter]}` : ''}
        </span>
      </div>

      {/* Table */}
      <table>
        <thead>
          <tr>
            <th style={thStyle} onClick={() => toggleSort('name')}>Command Name<SortIndicator col="name" /></th>
            <th style={thStyle} onClick={() => toggleSort('value')}>Value<SortIndicator col="value" /></th>
            {/* <th>Description</th> */}
            <th style={{ textAlign: 'center' }}>PM</th>
            <th style={{ textAlign: 'center' }}>PG4</th>
            <th style={{ textAlign: 'center' }}>BP2</th>
          </tr>
        </thead>
        <tbody>
          {filtered.map((row, i) => (
            <tr key={i}>
              <td style={{ whiteSpace: 'nowrap' }}><code>{row.name}</code></td>
              <td style={{ whiteSpace: 'nowrap' }}><code>{row.value}</code></td>
              {/* <td style={{ fontSize: '0.875rem' }}>{row.description}</td> */}
              <td style={{ textAlign: 'center' }}><Check active={row.pm} /></td>
              <td style={{ textAlign: 'center' }}><Check active={row.pg4} /></td>
              <td style={{ textAlign: 'center' }}><Check active={row.bp2} /></td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

Network Management (NMT) is a CANopen protocol for sending commands to modules, changing their operational states, and managing overall network behavior. The effect of these commands is to change the mode of the device (or devices collectively on the network). The command table is maintained as a list of `#defines` in the CanFest code base. Each module in the COSMIIC system maintains its own version of `def.h` in the application source code for each of the modules at `canFest/include/def.h`, and not all NMT commands are implemented by every module.

---

## Triggering an NMT Command

Users can control the use of NMT commands through NNP-API on MATLAB and using .nnpscripts. NMT commands must specify the target node, the specific hex code of the command, and relevant parameters.

- **`node`**: Target node ID (`0` = broadcast, `7` = PM, `1`–`15` = RM)
- **`command`**: NMT command as a hex string (e.g., `'95'`) or decimal integer
- **`param1`**, **`param2`** *(optional)*: Additional parameters (0–255)

### NMT in NNP-API

The [**NNP-API**](../Software/MATLAB-Interface/NNP-API.md) MATLAB library provides the `nmt` function in `NNPCORE` for sending NMT commands directly to any node on the network.

```matlab
dataOut = nnp.nmt(node, command)
dataOut = nnp.nmt(node, command, param1)
dataOut = nnp.nmt(node, command, param1, param2)
```

**Examples:**

```matlab
nnp.nmt(7, '95')       % Turn network ON (PM)
nnp.nmt(7, '8D', 1)    % Run script ID 1 on PM
```

### NMT in .nnpscript

Within a `.nnpscript` file, use the `NMT0`, `NMT1`, or `NMT2` opcodes depending on the number of parameters required.

```nnpscript
NMT0 <NODE> <CMD> 
NMT1 <NODE> <CMD> <PARAM1>
NMT2 <NODE> <CMD> <PARAM1> <PARAM2>
```

**Examples:**
